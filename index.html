// "__" means compiler builtin

Basic = __import!() // uses __name
i32 = __type!(enum.INTEGER, bits=32)

x = 5
y = 10

f = (z: i32) i32 {
  return x + y + z
}

Basic.assert!(f!12 == 27)

early_return = () {
  if x == 6 return {} // void value

  y := 7 // reassignment
  __parent.y := 8 // explicit scope

  // nested scopes
  {
    q = f!(x + 10)
    using Basic
    assert!(q == 30)
  }
}

Compiler = __import!()
Compiler.compile_to_file!(
  entry=early_return,
  name="does_nothing",
  optimize=enum.NONE,
)

// builtins are regular values
import = __import

// shadowed variables (new types)
x = import!"some_module"
y = import!()

bool = __type!(enum.BOOL)
true = 1.(bool)
false = 0.(bool)

toggle = false

w = (y: ^i32, r: []i32) i32 {
  if y.^ == 42 toggle := toggle.!
  sum = 0
  for r {
    sum += it + it_index
  }
  return sum
}

// defined in itself.

return = (expr=__expr) #(macro=true) {
  __caller.__return!(expr)
}

while = (cond=__expr, body=__expr) #(macro=true) {
  __loop!{
    __if!(cond.!, __break!())
    __mixin!body
  }
}

for = (iter=__expr, body=__expr) #(macro=true) {
  it_index = 0
  __loop!{
    __if!(it_index >= iter.count, __break!())
    it = iter.[it_index]
    __mixin!body
    it_index += 1
  }
}
